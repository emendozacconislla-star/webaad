<script>

let inventario = [];
let editando = -1;

function renderTabla(){

let tbody =
document.getElementById("tablaProductos");

tbody.innerHTML="";

inventario.forEach((p,index)=>{

tbody.innerHTML += `
<tr>

<td>${index+1}</td>

<td>${p.modelo}</td>

<td>${p.caracteristicas}</td>

<td>S/. ${p.precio}</td>

<td>${p.stock}</td>

<td>${p.existentes}</td>

<td>${p.vendidas}</td>

<td>${p.danadas}</td>

<td>

<button
class="btn-warning"
onclick="editar(${index})">
Editar
</button>

<button
class="btn-danger"
onclick="eliminar(${index})">
Eliminar
</button>

</td>

</tr>
`;

});

}

function obtenerFormulario(){

return{

modelo:
document.getElementById("modelo").value,

caracteristicas:
document.getElementById("caracteristicas").value,

precio:
document.getElementById("precio").value,

stock:
document.getElementById("stock").value,

existentes:
document.getElementById("existentes").value,

vendidas:
document.getElementById("vendidas").value,

danadas:
document.getElementById("danadas").value

};

}

function limpiar(){

document.querySelectorAll("input").forEach(i=>{

if(i.type!=="file")
i.value="";

});

editando=-1;

}

function guardar(){

let datos = obtenerFormulario();

if(!datos.modelo){

Swal.fire(
'Advertencia',
'Ingrese modelo y marca',
'warning'
);

return;
}

if(editando==-1){

inventario.push(datos);

Swal.fire(
'Correcto',
'Producto registrado',
'success'
);

}else{

inventario[editando]=datos;

Swal.fire(
'Actualizado',
'Registro actualizado',
'success'
);

}

renderTabla();
limpiar();

}

function editar(index){

let p=inventario[index];

document.getElementById("modelo").value=p.modelo;
document.getElementById("caracteristicas").value=p.caracteristicas;
document.getElementById("precio").value=p.precio;
document.getElementById("stock").value=p.stock;
document.getElementById("existentes").value=p.existentes;
document.getElementById("vendidas").value=p.vendidas;
document.getElementById("danadas").value=p.danadas;

editando=index;

}

function eliminar(index){

Swal.fire({

title:'¿Eliminar producto?',
text:'No podrá recuperarse',
icon:'warning',
showCancelButton:true,
confirmButtonText:'Eliminar'

}).then((result)=>{

if(result.isConfirmed){

inventario.splice(index,1);

renderTabla();

Swal.fire(
'Eliminado',
'Registro eliminado',
'success'
);

}

});

}

function buscarProducto(){

let filtro=
document.getElementById("buscar")
.value.toLowerCase();

let filas=
document.querySelectorAll("#tablaProductos tr");

filas.forEach(fila=>{

fila.style.display=
fila.innerText.toLowerCase()
.includes(filtro)
? ""
: "none";

});

}

function cargarExcel(event){

let archivo =
event.target.files[0];

if(!archivo) return;

let lector = new FileReader();

lector.onload=(e)=>{

let data =
new Uint8Array(e.target.result);

let workbook =
XLSX.read(data,{type:'array'});

let hoja =
workbook.Sheets[
workbook.SheetNames[0]
];

let json =
XLSX.utils.sheet_to_json(
hoja,
{header:1}
);

inventario=[];

for(let i=1;i<json.length;i++){

let fila=json[i];

if(fila.length>0){

inventario.push({

modelo:fila[0]||'',
caracteristicas:fila[1]||'',
precio:fila[2]||'',
stock:fila[3]||'',
existentes:fila[4]||'',
vendidas:fila[5]||'',
danadas:fila[6]||''

});

}

}

renderTabla();

Swal.fire(
'Excel cargado',
'Datos importados correctamente',
'success'
);

};

lector.readAsArrayBuffer(archivo);

}

function exportarExcel(){

let datos=[];

datos.push([
'Modelo y Marca',
'Características',
'Precio',
'Stock',
'Unidades Existentes',
'Unidades Vendidas',
'Unidades Dañadas'
]);

inventario.forEach(p=>{

datos.push([
p.modelo,
p.caracteristicas,
p.precio,
p.stock,
p.existentes,
p.vendidas,
p.danadas
]);

});

let ws =
XLSX.utils.aoa_to_sheet(datos);

let wb =
XLSX.utils.book_new();

XLSX.utils.book_append_sheet(
wb,
ws,
'Inventario'
);

let fecha =
new Date();

let nombre =
"inventarios-" +
fecha.getFullYear()+"-" +
String(fecha.getMonth()+1).padStart(2,'0')+"-" +
String(fecha.getDate()).padStart(2,'0')+"-" +
String(fecha.getHours()).padStart(2,'0')+"-" +
String(fecha.getMinutes()).padStart(2,'0')+"-" +
String(fecha.getSeconds()).padStart(2,'0') +
".xlsx";

XLSX.writeFile(
wb,
nombre
);

Swal.fire(
'Exportado',
'Copia de seguridad generada',
'success'
);

}
